<script>
var game2048 = (function() {
  var size = 4;
  var grid = [];
  var score = 0;
  var best = parseInt(localStorage.getItem('best2048') || '0');
  var moving = false;

  // OpenCode-style tile colors: blue-tinted terminal palette
  var tileColors = {
    2:    { bg:'rgba(37,99,235,0.08)',  fg:'var(--oc-blue-lighter)', border:'rgba(37,99,235,0.15)' },
    4:    { bg:'rgba(37,99,235,0.14)',  fg:'var(--oc-blue-lighter)', border:'rgba(37,99,235,0.2)' },
    8:    { bg:'rgba(37,99,235,0.22)',  fg:'#fff',                  border:'rgba(37,99,235,0.3)' },
    16:   { bg:'rgba(59,130,246,0.3)',  fg:'#fff',                  border:'rgba(59,130,246,0.4)' },
    32:   { bg:'rgba(96,165,250,0.35)', fg:'#fff',                  border:'rgba(96,165,250,0.45)' },
    64:   { bg:'rgba(96,165,250,0.45)', fg:'#fff',                  border:'rgba(96,165,250,0.55)' },
    128:  { bg:'rgba(34,197,94,0.2)',   fg:'#4ADE80',               border:'rgba(34,197,94,0.3)' },
    256:  { bg:'rgba(34,197,94,0.3)',   fg:'#4ADE80',               border:'rgba(34,197,94,0.4)' },
    512:  { bg:'rgba(34,197,94,0.4)',   fg:'#fff',                  border:'rgba(34,197,94,0.5)' },
    1024: { bg:'rgba(245,158,11,0.25)', fg:'#FBBF24',               border:'rgba(245,158,11,0.35)' },
    2048: { bg:'rgba(245,158,11,0.4)',  fg:'#fff',                  border:'rgba(245,158,11,0.5)' },
  };

  function init() {
    document.getElementById('best-2048').textContent = best;
    restart();
    bindEvents();
  }

  function restart() {
    grid = [];
    for (var r = 0; r < size; r++) {
      grid[r] = [];
      for (var c = 0; c < size; c++) grid[r][c] = 0;
    }
    score = 0;
    moving = false;
    document.getElementById('score-2048').textContent = '0';
    document.getElementById('overlay-2048').style.display = 'none';
    addRandom();
    addRandom();
    render();
  }

  function addRandom() {
    var empty = [];
    for (var r = 0; r < size; r++)
      for (var c = 0; c < size; c++)
        if (grid[r][c] === 0) empty.push({r:r, c:c});
    if (empty.length === 0) return;
    var cell = empty[Math.floor(Math.random() * empty.length)];
    grid[cell.r][cell.c] = Math.random() < 0.9 ? 2 : 4;
    cell._new = true;
    return cell;
  }

  function render(newCell, mergedCells) {
    var board = document.getElementById('board-2048');
    var html = '<div class="g2048-grid">';
    for (var r = 0; r < size; r++)
      for (var c = 0; c < size; c++)
        html += '<div class="g2048-cell"></div>';
    html += '</div><div class="g2048-tiles">';

    for (var r = 0; r < size; r++) {
      for (var c = 0; c < size; c++) {
        var v = grid[r][c];
        if (v === 0) continue;
        var col = tileColors[v] || { bg:'rgba(255,255,255,0.5)', fg:'#fff', border:'rgba(255,255,255,0.6)' };
        var isNew = newCell && newCell.r === r && newCell.c === c;
        var isMerged = mergedCells && mergedCells.some(function(m){ return m.r===r && m.c===c; });
        var cls = 'g2048-tile';
        if (isNew) cls += ' tile-new';
        if (isMerged) cls += ' tile-merged';
        var fs = v <= 64 ? '1.8em' : v <= 512 ? '1.5em' : v <= 8192 ? '1.2em' : '0.95em';
        var cellPct = 21.875;
        var step = cellPct + 2.5;
        html += '<div class="' + cls + '" style="' +
          'top:' + (2.5 + r * step) + '%;' +
          'left:' + (2.5 + c * step) + '%;' +
          'width:' + cellPct + '%;height:' + cellPct + '%;' +
          'background:' + col.bg + ';color:' + col.fg + ';' +
          'border-color:' + col.border + ';' +
          'font-size:' + fs + ';">' + v + '</div>';
      }
    }
    html += '</div>';
    board.innerHTML = html;
  }

  function move(dir) {
    if (moving) return;
    var moved = false;
    var mergedCells = [];
    var vectors = {up:{r:-1,c:0}, down:{r:1,c:0}, left:{r:0,c:-1}, right:{r:0,c:1}};
    var vec = vectors[dir];
    var traversals = buildTraversals(vec);

    var merged = [];
    for (var r = 0; r < size; r++) {
      merged[r] = [];
      for (var c = 0; c < size; c++) merged[r][c] = false;
    }

    traversals.r.forEach(function(r) {
      traversals.c.forEach(function(c) {
        if (grid[r][c] === 0) return;
        var val = grid[r][c];
        var farthest = findFarthest(r, c, vec, merged);
        if (farthest.merge) {
          grid[farthest.r][farthest.c] *= 2;
          grid[r][c] = 0;
          score += grid[farthest.r][farthest.c];
          merged[farthest.r][farthest.c] = true;
          mergedCells.push({r: farthest.r, c: farthest.c});
          moved = true;
        } else if (farthest.r !== r || farthest.c !== c) {
          grid[farthest.r][farthest.c] = val;
          grid[r][c] = 0;
          moved = true;
        }
      });
    });

    if (!moved) return;

    moving = true;
    document.getElementById('score-2048').textContent = score;
    if (score > best) {
      best = score;
      localStorage.setItem('best2048', best);
      document.getElementById('best-2048').textContent = best;
    }

    render(null, mergedCells);
    setTimeout(function() {
      var nc = addRandom();
      render(nc, []);
      moving = false;
      if (!movesAvailable()) {
        document.getElementById('overlay-msg-2048').textContent = 'GAME OVER';
        document.getElementById('overlay-2048').style.display = 'flex';
      }
    }, 130);
  }

  function buildTraversals(vec) {
    var traversals = { r: [], c: [] };
    for (var i = 0; i < size; i++) {
      traversals.r.push(i);
      traversals.c.push(i);
    }
    if (vec.r === 1) traversals.r.reverse();
    if (vec.c === 1) traversals.c.reverse();
    return traversals;
  }

  function findFarthest(r, c, vec, merged) {
    var prev;
    do {
      prev = {r: r, c: c};
      r += vec.r;
      c += vec.c;
    } while (withinBounds(r, c) && grid[r][c] === 0);

    if (withinBounds(r, c) && grid[r][c] === grid[prev.r][prev.c] && !merged[r][c]) {
      return { r: r, c: c, merge: true };
    }
    return { r: prev.r, c: prev.c, merge: false };
  }

  function withinBounds(r, c) {
    return r >= 0 && r < size && c >= 0 && c < size;
  }

  function movesAvailable() {
    for (var r = 0; r < size; r++)
      for (var c = 0; c < size; c++) {
        if (grid[r][c] === 0) return true;
        if (c < size - 1 && grid[r][c] === grid[r][c+1]) return true;
        if (r < size - 1 && grid[r][c] === grid[r+1][c]) return true;
      }
    return false;
  }

  function bindEvents() {
    document.addEventListener('keydown', function(e) {
      var map = { ArrowUp:'up', ArrowDown:'down', ArrowLeft:'left', ArrowRight:'right',
                  w:'up', s:'down', a:'left', d:'right' };
      if (map[e.key]) {
        e.preventDefault();
        move(map[e.key]);
      }
    });

    var startX, startY;
    var board = document.getElementById('board-2048');
    board.addEventListener('touchstart', function(e) {
      startX = e.touches[0].clientX;
      startY = e.touches[0].clientY;
    }, { passive: true });
    board.addEventListener('touchend', function(e) {
      if (!startX || !startY) return;
      var dx = e.changedTouches[0].clientX - startX;
      var dy = e.changedTouches[0].clientY - startY;
      var absDx = Math.abs(dx), absDy = Math.abs(dy);
      if (Math.max(absDx, absDy) < 20) return;
      if (absDx > absDy) move(dx > 0 ? 'right' : 'left');
      else move(dy > 0 ? 'down' : 'up');
      startX = startY = null;
    }, { passive: true });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

  return { restart: restart };
})();
</script>
